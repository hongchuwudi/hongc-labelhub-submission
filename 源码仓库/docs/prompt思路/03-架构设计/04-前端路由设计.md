# LabelHub 前端路由设计

## 1. 路由树

```
<Routes>
  {/* 公共布局 — 无登录守卫 */}
  <Route element={<AppLayout />}>
    <Route path="/"         element={<Home />} />
    <Route path="/login"    element={<Login />} />
    <Route path="/register" element={<Register />} />
  </Route>

  {/* 工作台布局 — 需登录 */}
  <Route element={<ProtectedRoute />}>
    <Route element={<WorkspaceLayout />}>
      
      {/* Owner 专属 */}
      <Route element={<RoleRoute roles={["owner"]} />}>
        <Route path="/datasets"       element={<Datasets />} />
        <Route path="/schemas"        element={<SchemaDesigner />} />
        <Route path="/tasks"          element={<TaskManage />} />
        <Route path="/ai-configs"     element={<AiConfigs />} />
        <Route path="/dashboard"      element={<Dashboard />} />
        <Route path="/export-history" element={<ExportHistory />} />
      </Route>

      {/* Owner + Reviewer 共享 */}
      <Route element={<RoleRoute roles={["owner", "reviewer"]} />}>
        <Route path="/review"   element={<Review />} />
        <Route path="/ai-review" element={<AiReview />} />
      </Route>

      {/* Labeler 专属 */}
      <Route element={<RoleRoute roles={["labeler"]} />}>
        <Route path="/labeling"  element={<Labeling />} />
        <Route path="/earnings"  element={<Earnings />} />
      </Route>

    </Route>
  </Route>

  {/* 404 */}
  <Route path="*" element={<NotFound />} />
</Routes>
```

## 2. 路由守卫组件

```tsx
// ProtectedRoute.tsx — 登录守卫
function ProtectedRoute() {
  const { isLoggedIn, authReady } = useStore();
  
  if (!authReady) return <Spin size="large" />;  // 初始化中
  if (!isLoggedIn) return <Navigate to="/login" replace />;
  
  return <Outlet />;
}

// RoleRoute.tsx — 角色守卫
function RoleRoute({ roles }: { roles: string[] }) {
  const { user } = useStore();
  
  if (!roles.includes(user.role)) {
    return <Result status="403" title="无权限访问" />;
  }
  
  return <Outlet />;
}
```

## 3. 布局组件

**AppLayout（公共布局）**：
```
┌────────────────────────────┐
│  TopHeader (variant="public") │
│  Logo + 工作台按钮          │
├────────────────────────────┤
│                            │
│        <Outlet />          │  ← 公开页内容
```

**WorkspaceLayout（工作台布局）**：
```

│          │        <Outlet />          │

└──────────┴───────────────────────────┘
```

## 4. 404 处理

```tsx
// 两种 404 场景
// 1. 路径不存在 → <Route path="*" element={<NotFound />} />
// 2. 角色不对 → RoleRoute 返回 <Result status="403" />

function NotFound() {
  return (
    <Result
      status="404"
      title="页面不存在"
      extra={<Button onClick={() => navigate("/")}>返回首页</Button>}
    />
  );
}
```

## 5. 登录后跳转逻辑

```tsx
// 登录成功后根据角色跳转
const roleHomeMap: Record<string, string> = {
  owner: "/datasets",
  labeler: "/labeling", 
  reviewer: "/review",
};

// 在 Login 组件提交成功后
const home = roleHomeMap[response.user.role] || "/";
navigate(home, { replace: true });
```
