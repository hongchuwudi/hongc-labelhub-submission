### Redis 客户端 (`infra/redis_client.py`)

**连接池管理**

```python
import redis

redis_pool = redis.ConnectionPool(
    host=os.getenv("REDIS_HOST", "localhost"),
    port=int(os.getenv("REDIS_PORT", 6379)),
    db=int(os.getenv("REDIS_DB", 0)),
    max_connections=20,
    decode_responses=True
)

def get_redis() -> redis.Redis:
    return redis.Redis(connection_pool=redis_pool)
```

**RedisLock — 分布式锁（SETNX + TTL）**

```python
class RedisLock:
    def __init__(self, redis_conn: redis.Redis, key: str, ttl: int = 10):
        self.redis = redis_conn
        self.key = f"lock:{key}"
        self.ttl = ttl

    def __enter__(self):
        acquired = self.redis.set(self.key, "1", nx=True, ex=self.ttl)
        if not acquired:
            raise ConflictException("resource is locked, please retry")
        return self

    def __exit__(self, *args):
        self.redis.delete(self.key)
```

使用 `with RedisLock(redis, "task:123")` 语法保护并发操作，自动获取并释放。

**RedisQueue — 阻塞队列（BLPOP）**

```python
class RedisQueue:
    def __init__(self, redis_conn: redis.Redis, queue_name: str):
        self.redis = redis_conn
        self.key = f"queue:{queue_name}"

    def push(self, value: str):
        self.redis.lpush(self.key, value)

    def pop(self, timeout: int = 0) -> Optional[str]:
        result = self.redis.brpop(self.key, timeout=timeout)
        return result[1] if result else None
```

**RedisCache — 带 TTL 的缓存**

```python
class RedisCache:
    def __init__(self, redis_conn: redis.Redis, prefix: str = "cache"):
        self.redis = redis_conn
        self.prefix = prefix

    def get(self, key: str) -> Optional[str]:
        return self.redis.get(f"{self.prefix}:{key}")

    def set(self, key: str, value: str, ttl: int = 300):
        self.redis.setex(f"{self.prefix}:{key}", ttl, value)

    def delete(self, key: str):
        self.redis.delete(f"{self.prefix}:{key}")
```

### RabbitMQ 客户端 (`infra/mq_client.py`)

**线程安全连接（blocking + 线程本地存储）**

```python
import pika
import threading

_thread_local = threading.local()

def get_connection():
    if not hasattr(_thread_local, "connection") or _thread_local.connection.is_closed:
        creds = pika.PlainCredentials(
            os.getenv("RABBITMQ_USER", "guest"),
            os.getenv("RABBITMQ_PASS", "guest")
        )
        params = pika.ConnectionParameters(
            host=os.getenv("RABBITMQ_HOST", "localhost"),
            credentials=creds,
            heartbeat=600
        )
        _thread_local.connection = pika.BlockingConnection(params)
        _thread_local.channel = _thread_local.connection.channel()
        # 自动声明 durable 队列
        _thread_local.channel.queue_declare(queue="labelhub.ai.review", durable=True)
        _thread_local.channel.queue_declare(queue="labelhub.export", durable=True)
    return _thread_local.connection, _thread_local.channel
```

**发布消息**

```python
def publish(queue: str, body: str):
    _, channel = get_connection()
    channel.basic_publish(
        exchange="",
        routing_key=queue,
        body=body,
        properties=pika.BasicProperties(delivery_mode=2)  # 持久化
    )
```

**消费消息**

```python
def consume(queue: str, callback: Callable[[str], None]):
    _, channel = get_connection()
    
    def on_message(ch, method, properties, body):
        try:
            callback(body.decode())
            ch.basic_ack(delivery_tag=method.delivery_tag)
        except Exception:
            ch.basic_nack(delivery_tag=method.delivery_tag, requeue=True)

    channel.basic_qos(prefetch_count=1)
    channel.basic_consume(queue=queue, on_message_callback=on_message)
    channel.start_consuming()
```

两个业务队列：`labelhub.ai.review`（AI 审核任务）和 `labelhub.export`（数据导出任务）。

### 异常处理 (`infra/exceptions.py`)

```python
class AppException(Exception):
    """业务异常基类"""
    def __init__(self, message: str, status_code: int = 400):
        self.message = message
        self.status_code = status_code

class BadRequestException(AppException):
    def __init__(self, message: str = "Bad request"):
        super().__init__(message, 400)

class NotFoundException(AppException):
    def __init__(self, message: str = "Not found"):
        super().__init__(message, 404)

class ForbiddenException(AppException):
    def __init__(self, message: str = "Forbidden"):
        super().__init__(message, 403)

class ConflictException(AppException):
    def __init__(self, message: str = "Conflict"):
        super().__init__(message, 409)
```

### 请求中间件 (`infra/middleware.py`)

**RequestLogMiddleware**

```python
class RequestLogMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        start = time.time()
        response = await call_next(request)
        elapsed = time.time() - start
        logger.info(f"{request.method} {request.url.path} → {response.status_code} ({elapsed:.3f}s)")
        return response
```

### 全局异常处理器

在 `main.py` 中注册：

```python
@app.exception_handler(AppException)
async def app_exception_handler(request: Request, exc: AppException):
    return JSONResponse(
        status_code=exc.status_code,
        content={"success": False, "message": exc.message, "code": exc.status_code}
    )

@app.exception_handler(RequestValidationError)
async def validation_handler(request: Request, exc: RequestValidationError):
    errors = [f"{e['loc'][-1]}: {e['msg']}" for e in exc.errors()]
    return JSONResponse(
        status_code=422,
        content={"success": False, "message": "; ".join(errors), "code": 422}
    )

@app.exception_handler(Exception)
async def global_exception_handler(request: Request, exc: Exception):
    logger.exception("Unhandled exception")
    return JSONResponse(
        status_code=500,
        content={"success": False, "message": "Internal server error", "code": 500}
    )

# 注册中间件
app.add_middleware(RequestLogMiddleware)
```

统一响应格式：`{ success: bool, message: string, code: int }`，所有异常层都遵循此格式。
